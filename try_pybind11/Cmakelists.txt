cmake_minimum_required(VERSION 3.20)

project(libimgui CXX)

set(CMAKE_CXX_STANDARD 20)

add_subdirectory(extern/pybind11)

add_subdirectory(extern/glfw)

include_directories(
    third_party/include/imgui
    third_party/include/imgui/backends
    src
)

file(GLOB_RECURSE SRC third_party/include/*.cpp src/*.cpp)

pybind11_add_module(libimgui ${SRC})

target_link_libraries(libimgui PRIVATE glfw opengl32)
